    \thmpara
    \subsection{Relation between Indicator Matrix and Weight}
    \begin{lemma}
        For any column-standard tableau \(S\), \(A'v'_{_S}\) is weight of \(S\).
    \end{lemma}
    \begin{proof}
        We prove by induction on number of columns \(m\) in tableau \(S\).
        For \(m=1\), let \(p\) be the column of \(S\), then \(v'_{_S}(q) = \delta_{pq}\).
        Thus we have, \(A'v'_{_S} = u_p\), which is exactly the weight of \(S\).

        Let \(S\) contain \(m+1\) columns, 
        we think of \(S\) as tableau obtained by appending a column \(p\) to another tableau \(S'\), 
        such \(S'\) and \(p\) exist. We have
        \[
            v'_{_S} = v'_{_{S'}} + v'_{p} 
            \implies A'v'_{_{S}} = A'v'_{_{S'}} + A'v'_{p} = wt(S') + wt(p) = wt(S).
        \]
    \end{proof}

    \begin{corollary}
        If a semi-standard tableau \(S\) is torus-invarient then \(v_{_S}\) is in the kernal of \(A\).
    \end{corollary}
    \begin{proof}
        Let \(S\) is torus-invarient of degree \(d\) then \(wt(S) = -dw\), thus we have, 
        \[
            Av_{_S} = [A'|w][v'^{t}_{_S}|d]^{t} = A'v'_{_S} + dw = wt(S) + dw = 0.
        \]
        Where notation \(x^t\) for vector \(x\) denotes transpose.
    \end{proof}

    \begin{corollary}
        Let \(x \ne 0\) be a non-negative integer vector in the kernal of \(A\)
        then there exist a torus-invarient semistandard tableau \(S\) of degree \(x_{l+1}\) 
        such that \(x = v_{_S}\).
    \end{corollary}
    \begin{proof}
        Define tableau \(S\) such that, \(p_i \in \SC\) appears \(x_i\) times. 
        Note since columns for \(S\) are from chain, \(S\) semistandard.
        It remains to show that degree of \(S\) is \(x_{l+1}\). 
        Given that, 
        \[
            Ax = 0 
            \implies [A'|w][v'^{t}_{_{S}}|x_{l+1}]^t = 0 
            \implies A'v'_{_{S}} = -x_{l+1}w
            \implies wt(S) = -x_{l+1}w.
        \]
        The notation \(x^t\) for vector \(x\) denotes transpose.
    \end{proof}
    \begin{corollary}
        The map in notation 5. is bijection:
        \begin{align*}
            STab(r,n,\SC) &\longrightarrow P_{\SC} \cap \IZ^{l+1} \setminus \{0\}\\
            T &\longmapsto v_{_T}\\
            T^v &\longmapsfrom v
        \end{align*}
    \end{corollary}
